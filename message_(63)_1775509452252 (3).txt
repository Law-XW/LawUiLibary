local dragging, dragInput, dragStart, startPos
local function update(input)
	local delta = input.Position - dragStart
	MobileFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end
MobileFrame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = MobileFrame.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)
MobileFrame.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)
UserInputService.InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		update(input)
	end
end)

-- Fly Logic
RunService.RenderStepped:Connect(function(dt)
	if not flightEnabled then
		lastPos = nil
		lastLook = nil
		return
	end

	local char = LP.Character
	if not char then return end

	local hum = char:FindFirstChildOfClass("Humanoid")
	if not hum or not hum.SeatPart then return end
	if hum.SeatPart.Name ~= "DriveSeat" then return end

	local seat = hum.SeatPart
	local vehicle = seat.Parent
	if not vehicle.PrimaryPart then
		vehicle.PrimaryPart = seat
	end

	local lookVector = Camera.CFrame.LookVector
	lastPos = lastPos or vehicle.PrimaryPart.Position
	lastLook = lastLook or lookVector

	local moveZ, moveY = 0, 0
	if UserInputService:IsKeyDown(Enum.KeyCode.W) then moveZ = 1 end
	if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveZ = -1 end
	if UserInputService:IsKeyDown(Enum.KeyCode.E) then moveY = 1 end
	if UserInputService:IsKeyDown(Enum.KeyCode.Q) then moveY = -1 end
	if UserInputService:IsKeyDown(Enum.KeyCode.W) or (seat:IsA("VehicleSeat") and seat.Throttle > 0) or mobileFwd then moveZ = 1 end
	if UserInputService:IsKeyDown(Enum.KeyCode.S) or (seat:IsA("VehicleSeat") and seat.Throttle < 0) or mobileBwd then moveZ = -1 end
	if UserInputService:IsKeyDown(Enum.KeyCode.E) or mobileUp then moveY = 1 end
	if UserInputService:IsKeyDown(Enum.KeyCode.Q) or mobileDown then moveY = -1 end

	local targetPos =
		vehicle.PrimaryPart.Position +
		(lookVector * moveZ * speed * dt) +
		(Vector3.new(0,1,0) * moveY * speed * dt)

	local newPos = lastPos:Lerp(targetPos, smoothPos)
	local smoothLook = lastLook:Lerp(lookVector, smoothRot)

	vehicle:SetPrimaryPartCFrame(
		CFrame.new(newPos, newPos + smoothLook)
	)

	lastPos = newPos
	lastLook = smoothLook

	for _, p in ipairs(vehicle:GetDescendants()) do
		if p:IsA("BasePart") then
			p.AssemblyLinearVelocity = Vector3.zero
			p.AssemblyAngularVelocity = Vector3.zero
		end
	end
end)

local function flyVehicleTo2(targetCFrame, callback)
	local player = game.Players.LocalPlayer
	local vehicle = workspace.Vehicles:FindFirstChild(player.Name)
	if not vehicle then return end

	local driveSeat = vehicle:FindFirstChild("DriveSeat")
	local humanoid = player.Character and player.Character:FindFirstChild("Humanoid")
	if humanoid and driveSeat then
		if not humanoid.SeatPart then
			driveSeat:Sit(humanoid)
		end
	end

	if not vehicle.PrimaryPart then
		vehicle.PrimaryPart = vehicle:FindFirstChild("Mass", true)
	end

	local startPos = vehicle.PrimaryPart.Position
	local targetPos = targetCFrame.Position

	-- Flughöhe während des Transports
	local flightHeight = -1

	-- Sofort auf Flughöhe -1 setzen
	local startFlightPos = Vector3.new(startPos.X, flightHeight, startPos.Z)
	vehicle:SetPrimaryPartCFrame(CFrame.new(startFlightPos))

	-- Zielposition für den Flug (auf Flughöhe -1)
	local flightTarget = Vector3.new(targetPos.X, flightHeight, targetPos.Z)

	-- Berechne Dauer für den Flug
	local distance = (Vector3.new(startPos.X, 0, startPos.Z) - Vector3.new(flightTarget.X, 0, flightTarget.Z)).Magnitude
	local duration = distance / FARMspeed

	-- Erstelle TweenInfo
	local info = TweenInfo.new(duration, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut)
	local CFrameValue = Instance.new("CFrameValue")
	CFrameValue.Value = vehicle:GetPrimaryPartCFrame()

	CFrameValue:GetPropertyChangedSignal("Value"):Connect(function()
		-- Während des Fluges Y-Koordinate auf -1 fixieren
		local currentPos = CFrameValue.Value.Position
		local fixedCFrame = CFrame.new(currentPos.X, flightHeight, currentPos.Z)
		vehicle:SetPrimaryPartCFrame(fixedCFrame)
		vehicle.PrimaryPart.Velocity = Vector3.new(0, 0, 0)
	end)

	local tween = TweenService:Create(CFrameValue, info, { Value = CFrame.new(flightTarget) })
	tween:Play()

	tween.Completed:Connect(function()
		CFrameValue:Destroy()
		-- Sobald angekommen, auf finale Höhe setzen
		vehicle:SetPrimaryPartCFrame(targetCFrame)

		if callback then callback() end
	end)
end

local function teleportToLocation(coordinates)
	ensurePlayerInVehicle()
	task.wait(0.5)
	flyVehicleTo2(coordinates)
end

local function FindDestination()
    -- Bus
    for _, v in pairs(Workspace.BusStops:GetDescendants()) do
        if v.Name == "SelectionBox" and v.Visible and v.Transparency ~= 1 then
            return v.Parent.CFrame
        end
    end
    -- Truck
    for _, v in pairs(Workspace.DeliveryDestinations:GetDescendants()) do
        if v.Name == "SelectionBox" and v.Visible and v.Transparency ~= 1 then
            return v.Parent.CFrame
        end
    end
    -- Fallback
    for _, v in pairs(LP.PlayerGui:GetDescendants()) do
        if v:IsA("BillboardGui") and v.Adornee and v.Adornee:IsA("BasePart") then
            return v.Adornee.CFrame
        end
    end
    return nil
end

local function TweenModel(model, targetCF, duration)
    if CurrentFarmTween then CurrentFarmTween:Cancel() end
    
    local CFrameValue = Instance.new("CFrameValue")
    CFrameValue.Value = model:GetPrimaryPartCFrame()
    
    CFrameValue:GetPropertyChangedSignal("Value"):Connect(function()
        if model.PrimaryPart then
            model:SetPrimaryPartCFrame(CFrameValue.Value)
            model.PrimaryPart.AssemblyLinearVelocity = Vector3.zero
            model.PrimaryPart.AssemblyAngularVelocity = Vector3.zero
        end
    end)
    
    local info = TweenInfo.new(duration, Enum.EasingStyle.Linear)
    local tween = TweenService:Create(CFrameValue, info, {Value = targetCF})
    CurrentFarmTween = tween
    tween:Play()
    
    local completed = false
    tween.Completed:Connect(function()
        completed = true
        CFrameValue:Destroy()
    end)
    
    repeat task.wait() until completed or StopFarm
    return completed
end

local function FarmStep()
    local veh = GetFarmVehicle()
    if not veh then
        LastFarmPos = nil
        return
    end
    
    if not veh.PrimaryPart then
        veh.PrimaryPart = veh:FindFirstChild("Mass") or veh:FindFirstChild("DriveSeat")
    end
    
    local dest = FindDestination()
    if dest then
        if LastFarmPos and (dest.Position - LastFarmPos.Position).Magnitude < 5 then
            return
        end
        
        LastFarmPos = dest
        
        local currentCF = veh:GetPrimaryPartCFrame()
        local downCF = CFrame.new(currentCF.X, currentCF.Y + FarmHeight, currentCF.Z)
        veh:SetPrimaryPartCFrame(downCF)
        
        if StopFarm then return end
        
        local targetUnder = dest * CFrame.new(0, FarmHeight, 0)
        local dist = (targetUnder.Position - downCF.Position).Magnitude
        local duration = dist / FarmSpeed
        
        if not TweenModel(veh, targetUnder, duration) then return end
        
        if StopFarm then return end
        
        local targetUp = dest
        TweenModel(veh, targetUp, 1)
        
        if not StopFarm then
             veh:SetPrimaryPartCFrame(veh:GetPrimaryPartCFrame() * CFrame.new(0, 0, -5))
        end
    else
        LastFarmPos = nil
    end
end

-- Returns true if a player has any form of wanted status
  local function isWantedPlayer(plr)
      if not plr.Character then return false end
      local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
      -- Check multiple attribute names that the game may use
      if hrp then
          for _, attrName in ipairs({"IsWanted", "Wanted", "WantedLevel", "WantedStars"}) do
              local v = hrp:GetAttribute(attrName)
              if v and v ~= false and v ~= 0 then return true end
          end
      end
      -- Also check for BoolValue / IntValue children on the character itself
      for _, valName in ipairs({"IsWanted", "Wanted", "WantedLevel", "WantedStars"}) do
          local obj = plr.Character:FindFirstChild(valName)
          if obj and obj.Value and obj.Value ~= false and obj.Value ~= 0 then return true end
      end
      return false
  end

  local function GetWantedPlayers()
      local wanted = {}
      for _, plr in pairs(Players:GetPlayers()) do
          if plr ~= LP and isWantedPlayer(plr) then
              table.insert(wanted, plr.Name)
          end
      end
      if #wanted == 0 then table.insert(wanted, "None") end
      return wanted
  end

-- Vortex Aimbot Functions
local function predictTarget(target, predictionTime)
	if not target.Character or not target.Character:FindFirstChild(AimPart) then return Vector3.zero end
	local pos = target.Character[AimPart].Position
	local vel = target.Character[AimPart].Velocity
	return pos + vel * predictionTime
end

local function getClosestTarget(ignoreFov)
	local cam = Workspace.CurrentCamera
	local closestPlayer = nil
	local closestDistance = math.huge
	local localPlayer = Players.LocalPlayer
	if not localPlayer.Character or not localPlayer.Character:FindFirstChild("HumanoidRootPart") then
		return nil
	end
	local localPosition = localPlayer.Character.HumanoidRootPart.Position
	local localTeam = localPlayer.Team

	for _, player in pairs(Players:GetPlayers()) do
		if player == localPlayer then continue end
		if not player.Character or not player.Character:FindFirstChild(AimPart) then continue end
		if WhitelistedUsers[player.Name] then continue end

		-- Ignore Untouchable
		if IgnoreUntouchable and player.Team and IgnoredTeams[player.Team.Name] then continue end

		-- Team Check
		if AimbotTeamCheck then
			if not localTeam or not player.Team then continue end
			if localTeam.Name == "Citizen" and player.Team.Name ~= "Police" then continue end
			if localTeam.Name == "Police" and player.Team.Name ~= "Citizen" then continue end
            if localTeam == player.Team then continue end
		end

		-- Ignore Not Wanted
		if IgnoreNotWanted then
			local isWanted = player.Character and player.Character:FindFirstChild("HumanoidRootPart") and 
				player.Character.HumanoidRootPart:GetAttribute("IsWanted") == true
			local isPolice = player.Team and player.Team.Name == "Police"

			if not isWanted and not isPolice then 
				continue
			end
		end

		-- Wall Check
		if AimbotWallCheck then
			local targetPartCheck = player.Character:FindFirstChild(AimPart)
			if targetPartCheck then
				local origin = cam.CFrame.Position
				local rayParams = RaycastParams.new()
				rayParams.FilterType = Enum.RaycastFilterType.Blacklist
				rayParams.FilterDescendantsInstances = {localPlayer.Character}
				local rayResult = Workspace:Raycast(origin, (targetPartCheck.Position - origin), rayParams)
				if rayResult and not rayResult.Instance:IsDescendantOf(player.Character) then continue end
			end
		end

		local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
		if not humanoid or humanoid.Health <= KnockThreshold then continue end

		local targetPosition = player.Character[AimPart].Position
		local distanceToTarget = (localPosition - targetPosition).Magnitude
		if distanceToTarget > MaxDistance then continue end

		local targetPos = cam:WorldToScreenPoint(targetPosition)
		if targetPos.Z < 0 then continue end

		local screenDistance = (Vector2.new(targetPos.X, targetPos.Y) - Vector2.new(cam.ViewportSize.X / 2, cam.ViewportSize.Y / 2)).Magnitude
		if not ignoreFov and screenDistance > FOVSize then continue end

		if screenDistance < closestDistance then
			closestDistance = screenDistance
			closestPlayer = player
		end
	end
	return closestPlayer
end

        return folder:FindFirstChild("7c113b14-5efb-4b43-bf60-fbc75c83d778")
    end
    return nil
end

-- Vortex Aimbot Logic
RunService.RenderStepped:Connect(function()
    local camV = Workspace.CurrentCamera
    local center2 = Vector2.new(camV.ViewportSize.X / 2, camV.ViewportSize.Y / 2)
    FOVCircle.Position = center2
	FOVCircle.Radius = FOVSize
    FOVCircle.Color = FOVColor
	FOVCircle.Visible = AimbotEnabled

    -- Tracer: line from screen center to nearest player in FOV
    if AimbotEnabled then
        local tracerTarget, minTD = nil, math.huge
        for _, plr in pairs(Players:GetPlayers()) do
            if plr ~= LP and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                local pos, onSc = camV:WorldToViewportPoint(plr.Character.HumanoidRootPart.Position)
                if onSc and pos.Z > 0 then
                    local pv2 = Vector2.new(pos.X, pos.Y)
                    local dist = (pv2 - center2).Magnitude
                    if dist < FOVSize and dist < minTD then minTD = dist; tracerTarget = pv2 end
                end
            end
        end
        if tracerTarget then
            TracerLine.From = center2; TracerLine.To = tracerTarget; TracerLine.Visible = true
        else TracerLine.Visible = false end
    else TracerLine.Visible = false end

    local isAiming = AimbotEnabled and (UserInputService:IsMouseButtonPressed(Enum.UserInputType.MouseButton2) or mobileAimActive)

    -- Sticky Aim: lock onto target on rising edge, release when done
    if mobileAimStyle == "Sticky" then
        if isAiming and not prevMobileAimActive then
            stickyTarget = getClosestTarget(true)  -- ignore FOV when locking
        elseif not isAiming then
            stickyTarget = nil
        end
    else
        stickyTarget = nil
    end
    prevMobileAimActive = isAiming

    local target = (mobileAimStyle == "Sticky" and stickyTarget) or getClosestTarget()

	if not target or not target.Character or not target.Character:FindFirstChild(AimPart) then return end

    if isAiming then
        local targetPos = PredictionEnabled and predictTarget(target, 0.2) or target.Character[AimPart].Position
        local current = cam.CFrame.LookVector
        local dir = (targetPos - cam.CFrame.Position).Unit
        local newLook = current:Lerp(dir, AimbotSmoothness)
        cam.CFrame = CFrame.new(cam.CFrame.Position, cam.CFrame.Position + newLook)
    end
end)

      local ok, d = pcall(Drawing.new, dType)
      if ok and d then return d end
      -- Fallback: silent no-op proxy for executors with limited Drawing support
      return setmetatable({}, {__index = function() return function() end end, __newindex = function() end})
  end

  local function RemoveESP(plr)
    if ESPDrawings[plr] then
        local esp = ESPDrawings[plr]
        if esp.Box then esp.Box:Remove() end
        if esp.Name then esp.Name:Remove() end
        if esp.Weapon then esp.Weapon:Remove() end
        if esp.Wanted then esp.Wanted:Remove() end
        if esp.Distance then esp.Distance:Remove() end
        if esp.HealthBarOutline then esp.HealthBarOutline:Remove() end
        if esp.HealthBar then esp.HealthBar:Remove() end
        if esp.Skeleton then
            for _, l in pairs(esp.Skeleton) do l:Remove() end
        end
        ESPDrawings[plr] = nil
    end
end

local function UpdateESP()
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LP then
            local success, _ = pcall(function()
                local char = plr.Character
                if char and char:FindFirstChild("HumanoidRootPart") and char:FindFirstChild("Humanoid") and ESPSettings.Enabled then
                    local isDead = char.Humanoid.Health <= 0
                    
                    -- Team Check (skip for dead — no team matters)
                    if not isDead and ESPSettings.TeamCheck and LP.Team and plr.Team and LP.Team == plr.Team then
                        if ESPDrawings[plr] then RemoveESP(plr) end
                        return
                    end

                    if not ESPDrawings[plr] then
                        ESPDrawings[plr] = {
                            Box = newDrawing("Square"),
                            Name = newDrawing("Text"),
                            Weapon = newDrawing("Text"),
                            Wanted = newDrawing("Text"),
                            Distance = newDrawing("Text"),
                            HealthBarOutline = newDrawing("Square"),
                            HealthBar = newDrawing("Square"),
                            Skeleton = {}
                        }
                        local d = ESPDrawings[plr]
                        d.Box.Visible = false; d.Box.Color = Color3.new(1, 0, 0); d.Box.Thickness = 1; d.Box.Filled = false
                        d.Name.Visible = false; d.Name.Color = Color3.new(1, 1, 1); d.Name.Size = 13; d.Name.Center = true; d.Name.Outline = true; d.Name.Font = 2
                        d.Weapon.Visible = false; d.Weapon.Color = Color3.new(1, 1, 1); d.Weapon.Size = 11; d.Weapon.Center = true; d.Weapon.Outline = true; d.Weapon.Font = 2
                        d.Wanted.Visible = false; d.Wanted.Color = Color3.fromRGB(255, 208, 0); d.Wanted.Size = 11; d.Wanted.Center = true; d.Wanted.Outline = true; d.Wanted.Font = 2
                        d.Distance.Visible = false; d.Distance.Color = Color3.new(1, 1, 1); d.Distance.Size = 11; d.Distance.Center = true; d.Distance.Outline = true; d.Distance.Font = 2
                        
                        d.HealthBarOutline.Visible = false; d.HealthBarOutline.Color = Color3.new(0,0,0); d.HealthBarOutline.Thickness = 1; d.HealthBarOutline.Filled = true
                        d.HealthBar.Visible = false; d.HealthBar.Color = Color3.new(0,1,0); d.HealthBar.Thickness = 1; d.HealthBar.Filled = true
                    end

                    local d = ESPDrawings[plr]
                    local hrp = char.HumanoidRootPart

                    -- Dead player: show black health bar, hide all other ESP elements
                    if isDead then
                        local vec, onSc = Camera:WorldToViewportPoint(hrp.Position)
                        if onSc and vec.Z > 0 and ESPSettings.Health then
                            local head = char:FindFirstChild("Head") or char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
                            local hp = head and Camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.5, 0))
                            local lp = Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0))
                            if hp and lp then
                                local h = math.max(1, lp.Y - hp.Y)
                                local w = h / 2
                                d.HealthBarOutline.Visible = true
                                d.HealthBarOutline.Size = Vector2.new(2, h + 2)
                                d.HealthBarOutline.Position = Vector2.new(vec.X - w / 2 - 5, hp.Y - 1)
                                d.HealthBar.Visible = true
                                d.HealthBar.Size = Vector2.new(1, h)
                                d.HealthBar.Position = Vector2.new(vec.X - w / 2 - 4.5, hp.Y)
                                d.HealthBar.Color = Color3.new(0, 0, 0)  -- BLACK when dead
                            else
                                d.HealthBar.Visible = false; d.HealthBarOutline.Visible = false
                            end
                        else
                            d.HealthBar.Visible = false; d.HealthBarOutline.Visible = false
                        end
                        d.Box.Visible = false; d.Name.Visible = false; d.Weapon.Visible = false
                        d.Wanted.Visible = false; d.Distance.Visible = false
                        for _, l in pairs(d.Skeleton) do l.Visible = false end
                        return
                    end

                    local vector, onScreen = Camera:WorldToViewportPoint(hrp.Position)

                    if onScreen and vector.Z > 0 then
                        local head = char:FindFirstChild("Head") or char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso")
                        local headPos = head and Camera:WorldToViewportPoint(head.Position + Vector3.new(0, 0.5, 0))
                        local legPos = Camera:WorldToViewportPoint(hrp.Position - Vector3.new(0, 3, 0))
                        
                        if headPos and legPos then
                            local height = legPos.Y - headPos.Y
                            local width = height / 2

                            -- Box
                            if ESPSettings.Box then
                                d.Box.Visible = true
                                d.Box.Size = Vector2.new(width, height)
                                d.Box.Position = Vector2.new(vector.X - width / 2, headPos.Y)
                                if plr.TeamColor then d.Box.Color = plr.TeamColor.Color else d.Box.Color = Color3.new(1,0,0) end
                            else
                                d.Box.Visible = false
                            end

                            -- Name
                            if ESPSettings.Name then
                                d.Name.Visible = true
                                d.Name.Text = plr.Name
                                d.Name.Position = Vector2.new(vector.X, headPos.Y - 18)
                                if plr.TeamColor then d.Name.Color = plr.TeamColor.Color end
                            else
                                d.Name.Visible = false
                            end

                            -- Wanted
                            if ESPSettings.Wanted then
                                local isWanted = char:FindFirstChild("HumanoidRootPart") and char.HumanoidRootPart:GetAttribute("IsWanted") == true
                                if isWanted then
                                    d.Wanted.Visible = true
                                    d.Wanted.Text = "WANTED"
                                    d.Wanted.Position = Vector2.new(vector.X, headPos.Y - 30)
                                else
                                    d.Wanted.Visible = false
                                end
                            else
                                d.Wanted.Visible = false
                            end

                            -- Weapon
                            if ESPSettings.Weapon then
                                local tool = char:FindFirstChildOfClass("Tool")
                                local weaponName = tool and tool.Name or "None"
                                d.Weapon.Visible = true
                                d.Weapon.Text = weaponName
                                d.Weapon.Position = Vector2.new(vector.X, legPos.Y + 5)
                            else
                                d.Weapon.Visible = false
                            end

                            -- Distance
                            if ESPSettings.Distance then
                                local dist = (Camera.CFrame.Position - hrp.Position).Magnitude
                                d.Distance.Visible = true
                                d.Distance.Text = math.floor(dist) .. "m"
                                d.Distance.Position = Vector2.new(vector.X, legPos.Y + 15)
                            else
                                d.Distance.Visible = false
                            end

                            -- Health
                            if ESPSettings.Health then
                                local health = char.Humanoid.Health
                                local maxHealth = char.Humanoid.MaxHealth
                                local healthPct = math.clamp(health / maxHealth, 0, 1)
                                local barHeight = height * healthPct

                                d.HealthBarOutline.Visible = true
                                d.HealthBarOutline.Size = Vector2.new(2, height + 2)
                                d.HealthBarOutline.Position = Vector2.new(vector.X - width / 2 - 5, headPos.Y - 1)

                                d.HealthBar.Visible = true
                                d.HealthBar.Size = Vector2.new(1, barHeight)
                                d.HealthBar.Position = Vector2.new(vector.X - width / 2 - 4.5, headPos.Y + height - barHeight)
                                d.HealthBar.Color = Color3.fromHSV(healthPct * 0.3, 1, 1) -- Red to Green
                            else
                                d.HealthBarOutline.Visible = false
                                d.HealthBar.Visible = false
                            end

                            -- Skeleton
                            if ESPSettings.Skeleton then
                                local joints = {
                                    {"Head", "UpperTorso"}, {"UpperTorso", "LowerTorso"}, 
                                    {"UpperTorso", "LeftUpperArm"}, {"LeftUpperArm", "LeftLowerArm"}, {"LeftLowerArm", "LeftHand"},
                                    {"UpperTorso", "RightUpperArm"}, {"RightUpperArm", "RightLowerArm"}, {"RightLowerArm", "RightHand"},
                                    {"LowerTorso", "LeftUpperLeg"}, {"LeftUpperLeg", "LeftLowerLeg"}, {"LeftLowerLeg", "LeftFoot"},
                                    {"LowerTorso", "RightUpperLeg"}, {"RightUpperLeg", "RightLowerLeg"}, {"RightLowerLeg", "RightFoot"}
                                }
                                if char.Humanoid.RigType == Enum.HumanoidRigType.R6 then
                                    joints = {{"Head", "Torso"}, {"Torso", "Left Arm"}, {"Torso", "Right Arm"}, {"Torso", "Left Leg"}, {"Torso", "Right Leg"}}
                                end

                                for i, pair in ipairs(joints) do
                                    local p1 = char:FindFirstChild(pair[1])
                                    local p2 = char:FindFirstChild(pair[2])
                                    if not d.Skeleton[i] then d.Skeleton[i] = NewLine() end
                                    local line = d.Skeleton[i]
                                    if p1 and p2 then
                                        local v1, s1 = Camera:WorldToViewportPoint(p1.Position)
                                        local v2, s2 = Camera:WorldToViewportPoint(p2.Position)
                                        if s1 and s2 then
                                            line.Visible = true; line.From = Vector2.new(v1.X, v1.Y); line.To = Vector2.new(v2.X, v2.Y)
                                            if plr.TeamColor then line.Color = plr.TeamColor.Color end
                                        else line.Visible = false end
                                    else line.Visible = false end
                                end
                            else
                                for _, l in pairs(d.Skeleton) do l.Visible = false end
                            end
                        end
                    else
                        d.Box.Visible = false; d.Name.Visible = false; d.Weapon.Visible = false; d.Wanted.Visible = false; d.Distance.Visible = false; d.HealthBarOutline.Visible = false; d.HealthBar.Visible = false
                        for _, l in pairs(d.Skeleton) do l.Visible = false end
                    end
                else
                    RemoveESP(plr)
                end
            end)
            if not success then
                RemoveESP(plr)
            end
        end
    end
end

end

local function CopySkin(target)
    if not target or not target.Character then return end
    local myChar = LP.Character
    if not myChar then return end

    for _, v in pairs(myChar:GetChildren()) do
        if v:IsA("Shirt") or v:IsA("Pants") or v:IsA("ShirtGraphic") or v:IsA("Accessory") or v:IsA("BodyColors") then
            v:Destroy()
        end
    end

    for _, v in pairs(target.Character:GetChildren()) do
        if v:IsA("Shirt") or v:IsA("Pants") or v:IsA("ShirtGraphic") or v:IsA("BodyColors") then
            v:Clone().Parent = myChar
        elseif v:IsA("Accessory") then
            v:Clone().Parent = myChar
        elseif v.Name == "Head" and v:FindFirstChild("face") then
            local myHead = myChar:FindFirstChild("Head")
            if myHead then
                if myHead:FindFirstChild("face") then myHead.face:Destroy() end
                v.face:Clone().Parent = myHead
            end
        end
    end
end

        end
    end

RunService.RenderStepped:Connect(function()
	if FullbrightEnabled then
		Lighting.Brightness = 2
		-- Only force 14:00 if Time Changer is NOT enabled
		if not TimeChangerEnabled then Lighting.ClockTime = 14 end
		Lighting.FogEnd = 100000
		Lighting.GlobalShadows = false
		Lighting.OutdoorAmbient = Color3.fromRGB(128, 128, 128)
	end
	if NoFogEnabled then
		Lighting.FogEnd = 1000000
	end
	if TimeChangerEnabled then
		Lighting.ClockTime = TimeChangerValue
	end
end)

RunService.RenderStepped:Connect(function(dt)
    if FreecamEnabled then
        local camCF = Camera.CFrame
        
        if FreecamMouseHeld then
            local delta = UserInputService:GetMouseDelta()
            local sens = 0.5
            local rot = Vector2.new(delta.X, delta.Y) * sens * dt * 60
            camCF = CFrame.new(camCF.Position) * CFrame.Angles(0, math.rad(-rot.X), 0) * camCF.Rotation * CFrame.Angles(math.rad(-rot.Y), 0, 0)
        end
        
        local move = Vector3.new()
        if FreecamKeys.W then move = move + Vector3.new(0, 0, -1) end
        if FreecamKeys.S then move = move + Vector3.new(0, 0, 1) end
        if FreecamKeys.A then move = move + Vector3.new(-1, 0, 0) end
        if FreecamKeys.D then move = move + Vector3.new(1, 0, 0) end
        if FreecamKeys.Q then move = move + Vector3.new(0, -1, 0) end
        if FreecamKeys.E then move = move + Vector3.new(0, 1, 0) end
        
        local speed = FreecamSpeed * 60 * dt
        if FreecamKeys.Shift then speed = speed * 5 end
        
        Camera.CFrame = camCF * CFrame.new(move * speed)
    end
end)

	Callback = function()
		local Http = game:GetService("HttpService")
		local TPS = game:GetService("TeleportService")
		local Api = "https://games.roblox.com/v1/games/"

		local _place = game.PlaceId
		local _servers = Api.._place.."/servers/Public?sortOrder=Desc&limit=100"
		
		local function ListServers(cursor)
			local Raw = game:HttpGet(_servers .. ((cursor and "&cursor="..cursor) or ""))
			return Http:JSONDecode(Raw)
		end

		local Next; repeat
			local Servers = ListServers(Next)
			for i,v in pairs(Servers.data) do
				if v.playing < v.maxPlayers and v.id ~= game.JobId then
					TPS:TeleportToPlaceInstance(_place, v.id, game.Players.LocalPlayer)
					return
				end
			end
			Next = Servers.nextPageCursor
		until not Next
	end
})

MiscTab:CreateSection("Safety")

MiscTab:CreateButton({
    Name = "❤️ Self Revive",
    Callback = function()
        local VirtualInputManager = game:GetService("VirtualInputManager")
        local startPosition = nil

        local function isPlayerDead()
            local player = game.Players.LocalPlayer
            if player and player.Character then
                local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
                if humanoid then
                    return humanoid.Health <= 24
                end
            end
            return false
        end

        local function goToHospitalAndSit()
            local player = game.Players.LocalPlayer
            local character = player.Character or player.CharacterAdded:Wait()

            character:MoveTo(Vector3.new(-107.427, 7.648, 1073.643))
            wait(1)

            local buildings = workspace:FindFirstChild("Buildings")
            if not buildings then return end
            local hospital = buildings:FindFirstChild("Hospital")
            if not hospital then return end
            local bed = hospital:FindFirstChild("HospitalBed")
            if not bed then return end
            local seat = bed:FindFirstChild("Seat")
            if not seat then return end

            character:MoveTo(seat.Position + Vector3.new(0, 2, 0))
            wait(0.7)

            local humanoid = character:FindFirstChildOfClass("Humanoid")
            if humanoid then
                seat:Sit(humanoid)
            end
        end

        local function pressSpace()
            VirtualInputManager:SendKeyEvent(true, Enum.KeyCode.Space, false, game)
            wait(0.1)
            VirtualInputManager:SendKeyEvent(false, Enum.KeyCode.Space, false, game)
            wait(0.2)
        end

        local player = game.Players.LocalPlayer

        if isPlayerDead() then
            startPosition = player.Character and player.Character.PrimaryPart and player.Character.PrimaryPart.CFrame or nil

            ensurePlayerInVehicle()
            wait(0.5)

            local hospitalCarPosition = CFrame.new(-89.70, 5.88, 1055.77)

            teleportToLocation(hospitalCarPosition, function()
                wait(1)
                player.Character:MoveTo(Vector3.new(-107.427, 7.648, 1073.643))
                wait(0.5)
                goToHospitalAndSit()

                task.spawn(function()
                    local humanoid = player.Character:FindFirstChildOfClass("Humanoid")
                    while humanoid and humanoid.Health <= 27 do
                        wait(1)
                        humanoid = player.Character:FindFirstChildOfClass("Humanoid") -- update
                    end

                    pressSpace()
                    wait(0.5)
                    ensurePlayerInVehicle()
                    wait(0.5)

                    if startPosition then
                        teleportToLocation(startPosition)
                    end
                end)
            end)
        else
            Rayfield:Notify({Title = "Self Revive", Content = "You are not dead.", Duration = 3})
        end
    end
})
