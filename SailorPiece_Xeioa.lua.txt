local function GetSafeModule(parent, name)
    if not parent then return nil end
    local obj = parent:FindFirstChild(name)
    if obj and obj:IsA("ModuleScript") then
        local ok, result = pcall(require, obj)
        if ok then return result end
    end
    return nil
end

local function Thread(featurePath, featureFunc, isEnabled, ...)
    local pathParts = featurePath:split(".")
    local currentTable = Flags
    for i = 1, #pathParts - 1 do
        local part = pathParts[i]
        if not currentTable[part] then currentTable[part] = {} end
        currentTable = currentTable[part]
    end
    local flagKey = pathParts[#pathParts]
    local activeThread = currentTable[flagKey]
    if isEnabled then
        if not activeThread or coroutine.status(activeThread) == "dead" then
            currentTable[flagKey] = task.spawn(featureFunc, ...)
        end
    else
        if activeThread and typeof(activeThread) == "thread" then
            task.cancel(activeThread)
            currentTable[flagKey] = nil
        end
    end
end

local function Cleanup(tbl)
    for key, value in pairs(tbl) do
        if typeof(value) == "RBXScriptConnection" then value:Disconnect(); tbl[key] = nil
        elseif typeof(value) == "thread" then task.cancel(value); tbl[key] = nil
        elseif type(value) == "table" then Cleanup(value) end
    end
end

local function Abbreviate(n)
    local abbrev = {{1e12,"T"},{1e9,"B"},{1e6,"M"},{1e3,"K"}}
    for _, v in ipairs(abbrev) do
        if n >= v[1] then return string.format("%.1f%s", n/v[1], v[2]) end
    end
    return tostring(n)
end

local function IsStrictBossMatch(npcName, displayName)
    if not npcName or not displayName then return false end
    local nLow = npcName:lower():gsub("%s+",""):gsub("boss$","")
    local dLow = displayName:lower():gsub("%s+",""):gsub("boss$","")
    return nLow:find(dLow, 1, true) ~= nil or dLow:find(nLow, 1, true) ~= nil
end

local function GetNearestIsland(targetPos, npcName)
    if npcName and Shared.BossTIMap[npcName] then return Shared.BossTIMap[npcName] end
    local nearest = "Starter"; local minDist = math.huge
    for islandName, crystal in pairs(IslandCrystals) do
        if crystal then
            local dist = (targetPos - crystal:GetPivot().Position).Magnitude
            if dist < minDist then minDist = dist; nearest = islandName end
        end
    end
    return nearest
end

local function IsValidTarget(npc)
    if not npc or not npc.Parent then return false end
    local hum = npc:FindFirstChildOfClass("Humanoid")
    if not hum then return false end
    if npc:FindFirstChild("IK_Active") then return true end
    local minMaxHP = tonumber(Flags_InstaKillMinHP) or 0
    local isEligible = Flags_InstaKill and hum.MaxHealth >= minMaxHP
    if isEligible then return (hum.Health > 0) or (npc == Shared.Target)
    else return hum.Health > 0 end
end

local function GetToolTypeFromModule(toolName)
    if not toolName then return "Melee" end
    local manual = {["Invisible"]="Power",["Bomb"]="Power",["Quake"]="Power"}
    if manual[toolName] then return manual[toolName] end
    if Modules.WeaponClass and Modules.WeaponClass.Tools then
        local data = Modules.WeaponClass.Tools[toolName]
        if data then return data.Type or "Melee" end
    end
    if toolName:find("Fruit") then return "Power" end
    return "Melee"
end

local function EquipWeapon()
    if not Remotes.EquipWeapon then return end
    local char = GetCharacter()
    if not char then return end
    local tool = char:FindFirstChildOfClass("Tool")
    if tool then return end
    local selectedTypes = Flags_SelectedWeaponType or {}
    local backpack = Plr:FindFirstChildOfClass("Backpack")
    if not backpack then return end
    for _, item in pairs(backpack:GetChildren()) do
        if item:IsA("Tool") then
            local tType = GetToolTypeFromModule(item.Name)
            if selectedTypes[tType] then
                if Remotes.EquipWeapon then
                    pcall(function() Remotes.EquipWeapon:FireServer(item.Name) end)
                end
                return
            end
        end
    end
    local anyTool = backpack:FindFirstChildOfClass("Tool")
    if anyTool and Remotes.EquipWeapon then
        pcall(function() Remotes.EquipWeapon:FireServer(anyTool.Name) end)
    end
end

local function IsSkillReady(key)
    local mainFrame = PGui:FindFirstChild("CooldownUI") and PGui.CooldownUI:FindFirstChild("MainFrame")
    if not mainFrame then return true end
    local slotMap = {Z=1,X=2,C=3,V=4,F=5}
    local slot = slotMap[key]
    if not slot then return true end
    for _, frame in pairs(mainFrame:GetChildren()) do
        if frame:IsA("Frame") then
            local idx = tonumber(frame.Name:match("%d+"))
            if idx == slot then
                local fill = frame:FindFirstChild("Fill") or frame:FindFirstChildWhichIsA("Frame", true)
                if fill and fill.Size.X.Scale < 0.05 then return true end
                return false
            end
        end
    end
    return true
end

local function GetCurrentPity()
    local pityUI = PGui:FindFirstChild("PityUI")
    if pityUI then
        local val = pityUI:FindFirstChildWhichIsA("TextLabel", true)
        if val then
            local cur, max = val.Text:match("(%d+)/(%d+)")
            if cur then return tonumber(cur), tonumber(max) end
        end
    end
    local pityAttr = Plr:GetAttribute("PityCount")
    local maxAttr = Plr:GetAttribute("PityMax") or 25
    return tonumber(pityAttr) or 0, tonumber(maxAttr) or 25
end

local function UpdateNPCLists()

    local seen = {}
    for _, npc in pairs(PATH.Mobs:GetChildren()) do
        if npc:IsA("Model") and npc:FindFirstChildOfClass("Humanoid") then

            if not seen[clean] and not npc.Name:find("Boss") then

            end

local function HybridMove(targetCF)
    local character = GetCharacter()
    local root = character and character:FindFirstChild("HumanoidRootPart")
    if not root then return end
    local distance = (root.Position - targetCF.Position).Magnitude
    local tweenSpeed = tonumber(Flags_TweenSpeed) or 160
    if distance > (tonumber(Flags_TargetDistTP) or 0) then
        local tweenTarget = targetCF * CFrame.new(0, 0, 150)
        local tweenDist = (root.Position - tweenTarget.Position).Magnitude
        local duration = tweenDist / tweenSpeed
        local tween = TweenService:Create(root, TweenInfo.new(duration, Enum.EasingStyle.Linear), {CFrame = tweenTarget})
        tween:Play(); tween.Completed:Wait()
        task.wait(0.1)
    end
    root.CFrame = targetCF
    root.AssemblyLinearVelocity = Vector3.new(0, 0.01, 0)
    task.wait(0.2)
end

local function ExecuteFarmLogic(target, island, farmType)
    local char = GetCharacter()
    local root = char and char:FindFirstChild("HumanoidRootPart")
    if not char or not target or Shared.Recovering or not root then return end
    if Shared.MovingIsland then return end
    Shared.Target = target

    if Flags_IslandTP then
        if island and island ~= "" and island ~= Shared.Island then
            Shared.MovingIsland = true
            if Remotes.TP_Portal then pcall(function() Remotes.TP_Portal:FireServer(island) end) end
            task.wait(tonumber(Flags_IslandTPCD) or 0.67)
            Shared.Island = island
            Shared.MovingIsland = false
            return
        end
    end

    local targetPivot = target:GetPivot()
    local targetPos = targetPivot.Position
    local distVal = tonumber(Flags_Distance) or 12
    local posType = Flags_FarmType or "Behind"

    local finalPos
    if posType == "Above" then
        finalPos = targetPos + Vector3.new(0, distVal, 0)
    elseif posType == "Below" then
        finalPos = targetPos + Vector3.new(0, -distVal, 0)
    else
        finalPos = (targetPivot * CFrame.new(0, 0, distVal)).Position
    end

    local finalDestination = CFrame.lookAt(finalPos, targetPos)

    if (root.Position - finalPos).Magnitude > 0.5 then
        if (Flags_MovementType or "Tween") == "Teleport" then
            root.CFrame = finalDestination
        else
            local distance = (root.Position - finalPos).Magnitude
            local speed = tonumber(Flags_TweenSpeed) or 160
            TweenService:Create(root, TweenInfo.new(distance/speed, Enum.EasingStyle.Linear), {CFrame = finalDestination}):Play()
        end
    end

    root.AssemblyLinearVelocity = Vector3.zero
    root.AssemblyAngularVelocity = Vector3.zero
end

local function GetBestMobCluster(mobNamesDictionary)
    local allMobs = {}
    if type(mobNamesDictionary) ~= "table" then return nil end
    for _, npc in pairs(PATH.Mobs:GetChildren()) do
        if npc:IsA("Model") and npc:FindFirstChildOfClass("Humanoid") then
            local cleanName = npc.Name:gsub("%d+$","")
            if mobNamesDictionary[cleanName] and IsValidTarget(npc) then
                table.insert(allMobs, npc)
            end
        end
    end
    if #allMobs == 0 then return nil end
    local bestMob = allMobs[1]; local maxNearby = 0
    for _, mobA in ipairs(allMobs) do
        local nearby = 0; local posA = mobA:GetPivot().Position
        for _, mobB in ipairs(allMobs) do
            if (posA - mobB:GetPivot().Position).Magnitude <= 35 then nearby += 1 end
        end
        if nearby > maxNearby then maxNearby = nearby; bestMob = mobA end
    end
    return bestMob, maxNearby
end

local function UpdateQuest()
    if not Flags_LevelFarm then return end
    local QuestModule = Modules.Quests
    local playerLevel = Plr.Data.Level.Value
    local bestNPC = "QuestNPC1"; local highestLevel = -1
    for npcId, questData in pairs(QuestModule.RepeatableQuests) do
        local reqLevel = questData.recommendedLevel or 0
        if playerLevel >= reqLevel and reqLevel > highestLevel then
            highestLevel = reqLevel; bestNPC = npcId
        end
    end

    local settings = PGui:FindFirstChild("SettingsUI")
    if settings then
        local tog1 = settings:FindFirstChild("Toggle_EnableQuestRepeat", true)
        if tog1 and tog1:FindFirstChild("SettingsHolder") and tog1.SettingsHolder:FindFirstChild("Off") and tog1.SettingsHolder.Off.Visible then
            if Remotes.SettingsToggle then pcall(function() Remotes.SettingsToggle:FireServer("EnableQuestRepeat", true) end) end
            task.wait(0.3)
        end
    end

    local questUI = PGui:FindFirstChild("QuestUI") and PGui.QuestUI:FindFirstChild("Quest")
    if Shared.QuestNPC ~= bestNPC or (questUI and not questUI.Visible) then
        if Remotes.QuestAbandon then pcall(function() Remotes.QuestAbandon:FireServer("repeatable") end) end
        task.wait(1)
        if Remotes.QuestAccept then pcall(function() Remotes.QuestAccept:FireServer(bestNPC) end) end
        task.wait(1)
        Shared.QuestNPC = bestNPC
    end
end

local function Func_AutoSkill()
    local keyToEnum = {Z=Enum.KeyCode.Z, X=Enum.KeyCode.X, C=Enum.KeyCode.C, V=Enum.KeyCode.V, F=Enum.KeyCode.F}
    local keyToSlot = {Z=1, X=2, C=3, V=4, F=5}
    local priority = {"Z","X","C","V","F"}
    while task.wait() do
        if not Flags_AutoSkill then break end
        local target = Shared.Target
        if Flags_OnlyTarget and (not Shared.Farm or not target or not target.Parent) then continue end
        local canExecute = true
        if Flags_AutoSkill_BossOnly then
            if not target or not target.Parent then canExecute = false
            else
                local npcHum = target:FindFirstChildOfClass("Humanoid")
                local isRealBoss = target.Name:find("Boss") and not table.find(MiniBossList, target.Name)
                local hpPercent = npcHum and (npcHum.Health / npcHum.MaxHealth * 100) or 101
                local threshold = tonumber(Flags_AutoSkill_BossHP) or 100
                if not isRealBoss or hpPercent > threshold then canExecute = false end
            end
        end
        if not canExecute then continue end
        local char = GetCharacter()
        local tool = char and char:FindFirstChildOfClass("Tool")
        if not tool then continue end
        local toolType = GetToolTypeFromModule(tool.Name)
        local selected = Flags_SelectedSkills or {}
        local useMode = Flags_AutoSkillType or "Normal"
        if useMode == "Instant" then
            for _, key in ipairs(priority) do
                if selected[key] then
                    if toolType == "Power" and Remotes.UseFruit then
                        pcall(function() Remotes.UseFruit:FireServer("UseAbility", {FruitPower=tool.Name:gsub(" Fruit",""), KeyCode=keyToEnum[key]}) end)
                    elseif Remotes.UseSkill then
                        pcall(function() Remotes.UseSkill:FireServer(keyToSlot[key]) end)
                    end
                end
            end
            task.wait(0.01)
        else
            for _, key in ipairs(priority) do
                if selected[key] and IsSkillReady(key) then
                    if toolType == "Power" and Remotes.UseFruit then
                        pcall(function() Remotes.UseFruit:FireServer("UseAbility", {FruitPower=tool.Name:gsub(" Fruit",""), KeyCode=keyToEnum[key]}) end)
                    elseif Remotes.UseSkill then
                        pcall(function() Remotes.UseSkill:FireServer(keyToSlot[key]) end)
                    end
                    task.wait(0.1); break
                end
            end
        end
    end
end

local function Func_AutoCombo()
    Shared.ComboIdx = 1
    while Flags_AutoCombo do
        task.wait(0.1)
        local rawPattern = Flags_ComboPattern or "Z > X > C > V > F"
        if rawPattern == "" then continue end
        Shared.ParsedCombo = {}
        for item in string.gmatch(rawPattern:upper():gsub("%s+",""), "([^,>]+)") do
            table.insert(Shared.ParsedCombo, item)
        end
        if #Shared.ParsedCombo == 0 then continue end
        if Shared.ComboIdx > #Shared.ParsedCombo then Shared.ComboIdx = 1 end
        if IsBusy() then
            local waitStart = tick()
            repeat task.wait(0.1) until not IsBusy() or (tick() - waitStart > 8)
        end
        task.wait(0.4)
        if Flags_ComboBossOnly then
            if not Shared.Target or not Shared.Target.Parent or not Shared.Target.Name:lower():find("boss") then
                Shared.ComboIdx = 1; task.wait(0.5); continue
            end
        end
        local currentAction = Shared.ParsedCombo[Shared.ComboIdx]
        local waitTime = tonumber(currentAction)
        if waitTime then
            if (Flags_ComboMode or "Normal") == "Normal" then task.wait(waitTime) end
            Shared.ComboIdx += 1; continue
        end
        if IsSkillReady(currentAction) then
            local isF = (currentAction == "F")
            if isF then
                if Remotes.UseSkill then
                    local uiConfirmed = false
                    repeat
                        EquipWeapon()
                        pcall(function() Remotes.UseSkill:FireServer(5) end)
                        local check = tick()
                        repeat task.wait(0.1)
                            if not IsSkillReady("F") then uiConfirmed = true end
                        until uiConfirmed or (tick() - check > 1.0)
                    until uiConfirmed or not Flags_AutoCombo
                    local ffStarted = false
                    local catchTimer = tick()
                    repeat task.wait()
                        if IsBusy() then ffStarted = true end
                    until ffStarted or (tick() - catchTimer > 2.0)
                    if ffStarted then
                        local holdStart = tick()
                        repeat task.wait(0.1) until not IsBusy() or (tick() - holdStart > 15)
                    else task.wait(2.5) end
                end
                Shared.ComboIdx += 1; task.wait(0.3)
            else
                local slot = ({Z=1,X=2,C=3,V=4})[currentAction] or 1
                if Remotes.UseSkill then
                    local stepDone = false
                    repeat
                        pcall(function() Remotes.UseSkill:FireServer(slot) end)
                        local check = tick()
                        repeat task.wait(0.1)
                            if not IsSkillReady(currentAction) or IsBusy() then stepDone = true end
                        until stepDone or (tick() - check > 1.2)
                    until stepDone or not Flags_AutoCombo
                    if stepDone then Shared.ComboIdx += 1; task.wait(0.2) end
                end
            end
        else task.wait(0.2) end
    end
end

                    end
                else
                    if Remotes.AddStat then pcall(function() Remotes.AddStat:FireServer(activeStats[1], availablePoints) end) end
                end
            end
        end
    end
end

    while Flags_AutoRollStats do

        local selectedRanks = Flags_SelectedRank or {}
        local hasStat = false; for _ in pairs(selectedStats) do hasStat = true; break end
        local hasRank = false; for _ in pairs(selectedRanks) do hasRank = true; break end
        if not hasStat or not hasRank then
            Library:MakeNotification({Name="Error", Content="Select at least one Stat and Rank!", Time=5})
            Flags_AutoRollStats = false; break
        end
        if not next(Shared.GemStats) then task.wait(0.1); continue end
        local workDone = true
        for _, statName in ipairs(Tables.GemStat) do
            if selectedStats[statName] then
                local currentData = Shared.GemStats[statName]
                if currentData then
                    local currentRank = currentData.Rank
                    if not selectedRanks[currentRank] then
                        workDone = false
                        if Remotes.RerollSingleStat then
                            pcall(function() Remotes.RerollSingleStat:InvokeServer(statName) end)
                        end
                        task.wait(tonumber(Flags_StatsRollCD) or 0.1)
                        break
                    end
                end
            end
        end
        if workDone then
            Library:MakeNotification({Name="Done", Content="Stats rolled successfully.", Time=5})
            Flags_AutoRollStats = false; break
        end
        task.wait()
    end
end

local function Func_UnifiedRollManager()
    while task.wait() do

                if Remotes.UseItem then pcall(function() Remotes.UseItem:FireServer("Use","Race Reroll",1) end) end
                task.wait(tonumber(Flags_RollCD) or 0.3)
            end
            continue
        end
        if Flags_AutoClan then
            local currentClan = Plr:GetAttribute("CurrentClan")
            local selected = Flags_SelectedClan or {}
            if selected[currentClan] then
                Library:MakeNotification({Name="Success!", Content="Got Clan: "..tostring(currentClan), Time=5})

                if Remotes.UseItem then pcall(function() Remotes.UseItem:FireServer("Use","Clan Reroll",1) end) end
                task.wait(tonumber(Flags_RollCD) or 0.3)

local function Func_AutoSkillTree()
    while Flags_AutoSkillTree do
        task.wait(0.5)
        if not next(Shared.SkillTree.Nodes) and (Shared.SkillTree.SkillPoints or 0) == 0 then continue end
        local points = Shared.SkillTree.SkillPoints or 0
        if points <= 0 then continue end
        local SkillTreeModule = GetSafeModule(RS:FindFirstChild("Modules"), "SkillTreeConfig")
        if SkillTreeModule and SkillTreeModule.Branches then
            for _, branch in pairs(SkillTreeModule.Branches) do
                for _, node in ipairs(branch.Nodes) do
                    if not Shared.SkillTree.Nodes[node.Id] then
                        if points >= node.Cost then
                            if Remotes.SkillTreeUpgrade then
                                pcall(function() Remotes.SkillTreeUpgrade:FireServer(node.Id) end)
                                Shared.SkillTree.SkillPoints -= node.Cost
                                task.wait(0.3)
                            end
                        end
                        break
                    end
                end
            end
        end
    end
end

local function GetSecondsFromTimer(text)
    local m, s = text:match("(%d+):(%d+)")
    if m then return tonumber(m) * 60 + tonumber(s) end
    local h, m2, s2 = text:match("(%d+):(%d+):(%d+)")
    if h then return tonumber(h) * 3600 + tonumber(m2) * 60 + tonumber(s2) end
    return nil
end

local function Func_AutoMerchant()
    local function SyncClock()
        OpenMerchantInterface(); task.wait(1)
        local MerchantUI = PGui:FindFirstChild("MerchantUI")
        if MerchantUI then
            local Label = MerchantUI:FindFirstChild("RefreshTimerLabel", true)
            if Label and Label.Text:find(":") then
                local serverSecs = GetSecondsFromTimer(Label.Text)
                if serverSecs then Shared.LocalMerchantTime = serverSecs end
            end
            if MerchantUI:FindFirstChild("MainFrame") then MerchantUI.MainFrame.Visible = false end
        end
    end

    SyncClock()

    local LastTimerText = ""
    while Flags_AutoMerchant do
        task.wait(1)
        local MerchantUI = PGui:FindFirstChild("MerchantUI")
        if not MerchantUI then continue end
        local Label = MerchantUI:FindFirstChild("RefreshTimerLabel", true)
        if Label and Label.Text ~= "" then
            local s = GetSecondsFromTimer(Label.Text)
            if s then
                Shared.LocalMerchantTime = s
                if Label.Text ~= LastTimerText then LastTimerText = Label.Text; Shared.LastTimerTick = tick() end
            else
                Shared.LocalMerchantTime = math.max(0, Shared.LocalMerchantTime - 1)
            end
        else
            Shared.LocalMerchantTime = math.max(0, Shared.LocalMerchantTime - 1)
        end

        local isRefresh = (Shared.LocalMerchantTime <= 1) or (Shared.LocalMerchantTime >= 1799)
        if not Shared.FirstMerchantSync or isRefresh then
            if not Shared.MerchantExecute then
                Shared.MerchantExecute = true
                task.spawn(function()
                    OpenMerchantInterface(); task.wait(2)
                    local Holder = MerchantUI:FindFirstChild("Holder", true)
                    if Holder then
                        local selectedItems = Flags_SelectedMerchantItems or {}
                        for _, child in pairs(Holder:GetChildren()) do
                            if child:IsA("Frame") and child.Name ~= "Item" then
                                if selectedItems[child.Name] and Remotes.MerchantBuy then
                                    pcall(function() Remotes.MerchantBuy:InvokeServer(child.Name, 99) end)
                                    task.wait(1.3)
                                end
                            end
                        end
                    end
                    if MerchantUI:FindFirstChild("MainFrame") then MerchantUI.MainFrame.Visible = false end
                    Shared.FirstMerchantSync = true
                    Shared.MerchantExecute = false
                end)
            end
        end
        if tick() - Shared.LastTimerTick > 30 then
            task.spawn(SyncClock)
            Shared.LastTimerTick = tick()
        end
    end
end

local function Func_AntiKnockback()
    local knockbackConns = {}

    local function ApplyAntiKB(character)
        if not character then return end
        local root = character:WaitForChild("HumanoidRootPart", 10)
        if root then
            local conn = root.ChildAdded:Connect(function(child)
                if not Flags_AntiKnockback then return end
                if child:IsA("BodyVelocity") and child.MaxForce == Vector3.new(40000,40000,40000) then
                    child:Destroy()
                end
            end)
            table.insert(knockbackConns, conn)
        end
    end

    if Plr.Character then ApplyAntiKB(Plr.Character) end
    local charAddedConn = Plr.CharacterAdded:Connect(function(newChar) ApplyAntiKB(newChar) end)
    table.insert(knockbackConns, charAddedConn)

    repeat task.wait(1) until not Flags_AntiKnockback

    for _, conn in pairs(knockbackConns) do if conn then conn:Disconnect() end end
end

local function PostToWebhook()
    local url = Flags_WebhookURL or ""
    if url == "" or not url:find("discord.com/api/webhooks/") then return end
    local selected = Flags_SelectedData or {}
    local data = Plr.Data
    local lstats = Plr:FindFirstChild("leaderstats")
    local bounty = lstats and lstats:FindFirstChild("Bounty") and lstats.Bounty.Value or 0
    local desc = "### SailorPiece | Xeioa\n"
    if selected["Name"] then
        desc = desc .. string.format("\n**Player:** ||%s||\n", Plr.Name)
    end
    if selected["Stats"] then
        desc = desc .. string.format("**Level:** `%s`\n", tostring(data.Level.Value))
        desc = desc .. string.format("**Currency:** %s | %s Gems\n", Abbreviate(data.Money.Value), tostring(data.Gems.Value))
        desc = desc .. string.format("**Bounty:** %s\n", Abbreviate(bounty))
    end
    local payload = {
        ["embeds"] = {{
            ["description"] = desc,
            ["color"] = tonumber("8a2be2", 16),
            ["footer"] = {["text"] = string.format("SailorPiece | Xeioa • %s", os.date("%x %X"))}
        }}
    }
    if Flags_PingUser then payload["content"] = (Flags_UID and Flags_UID ~= "" and "<@"..Flags_UID..">") or "@everyone" end
    task.spawn(function()
        local reqFunc = request or http_request
        if reqFunc then
            pcall(function()
                reqFunc({Url=url, Method="POST", Headers={["Content-Type"]="application/json"}, Body=HttpService:JSONEncode(payload)})
            end)
        end
    end)
end

local function CheckPlayerForSafety(targetPlayer)
    if not Flags_AutoKick then return end
    if targetPlayer == Plr then return end
    local kickTypes = Flags_KickType or {}
    if kickTypes["Player Join"] then
        task.wait(0.5)
        Plr:Kick("\n[SailorPiece | Xeioa]\nReason: A player joined (" .. targetPlayer.Name .. ")")
        return
    end
    if kickTypes["Mod"] then
        local ok, rank = pcall(function() return targetPlayer:GetRankInGroup(TargetGroupId) end)
        if ok and table.find(BannedRanks, rank) then
            task.wait(0.5)
            Plr:Kick("\n[SailorPiece | Xeioa]\nReason: Moderator Detected (" .. targetPlayer.Name .. ")")
        end
    end
end

local function CheckServerTypeSafety()
    if not Flags_AutoKick then return end
    local kickTypes = Flags_KickType or {}
    if kickTypes["Public Server"] then
        local ok, serverType = pcall(function()
            local remote = game:GetService("RobloxReplicatedStorage"):WaitForChild("GetServerType", 2)
            if remote then return remote:InvokeServer() end

            task.wait(0.8)
            Plr:Kick("\n[SailorPiece | Xeioa]\nReason: You are in a public server.")
        end
    end
end

task.spawn(function()
    while task.wait() do
        if not getgenv().SP_Xeioa_Running then break end
        if not Shared.Farm or Shared.MerchantBusy then Shared.Target = nil; continue end
        local char = GetCharacter()
        if not char or Shared.Recovering then continue end
        if Shared.TargetValid and (not Shared.Target or not Shared.Target.Parent) then
            Shared.KillTick = tick(); Shared.TargetValid = false
        end
        if tick() - Shared.KillTick < (tonumber(Flags_TargetTPCD) or 0) then continue end

        local t, isl, fType
        t, isl, fType = GetPityTarget()
        if not t then t, isl, fType = GetSummonTarget() end
        if not t then t, isl, fType = GetOtherTarget() end
        if not t then t, isl, fType = GetWorldBossTarget() end
        if not t then t, isl, fType = GetLevelFarmTarget() end
        if not t then t, isl, fType = GetAllMobTarget() end
        if not t then t, isl, fType = GetMobTarget() end
        if not t then t, isl, fType = GetAltHelpTarget() end

        if t then
            Shared.Target = t; Shared.TargetValid = true
            ExecuteFarmLogic(t, isl, fType)
        else
            Shared.Target = nil
        end
    end
end)

task.spawn(function()
    while task.wait() do
        if not getgenv().SP_Xeioa_Running then break end
        if Shared.AltActive or not Shared.Farm or Shared.MerchantBusy or not Shared.Target then continue end
        local char = GetCharacter(); local target = Shared.Target
        if not char or not target or not target.Parent then continue end
        local npcHum = target:FindFirstChildOfClass("Humanoid")
        local npcRoot = target:FindFirstChild("HumanoidRootPart")
        local root = char:FindFirstChild("HumanoidRootPart")
        if npcHum and npcRoot and root then
            local currentDist = (root.Position - npcRoot.Position).Magnitude
            local hpPercent = (npcHum.Health / npcHum.MaxHealth) * 100
            local minMaxHP = tonumber(Flags_InstaKillMinHP) or 0
            local ikThreshold = tonumber(Flags_InstaKillHP) or 90
            if Flags_InstaKill and npcHum.MaxHealth >= minMaxHP and hpPercent < ikThreshold then
                npcHum.Health = 0
                if not target:FindFirstChild("IK_Active") then
                    local tag = Instance.new("Folder")
                    tag.Name = "IK_Active"
                    tag:SetAttribute("TriggerTime", tick())
                    tag.Parent = target
                end
            end
            if currentDist < 35 then
                local m1Delay = tonumber(Flags_M1Speed) or 0.2
                if tick() - Shared.LastM1 >= m1Delay then
                    EquipWeapon()
                    if Remotes.M1 then pcall(function() Remotes.M1:FireServer() end) end
                    Shared.LastM1 = tick()
                end
            end
        end
    end
end)
